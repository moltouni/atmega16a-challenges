.equ LED_PORT = PORTA
.equ LED_DDR = DDRA
.equ BUTT_PORT = PORTB
.equ BUTT_DDR = DDRB
.equ cnt = 10

.def tmp = r16

.cseg
        rjmp reset

reset:
        ldi tmp, low(RAMEND)
        out SPL, tmp
        ldi tmp, high(RAMEND)
        out SPH, tmp

        ldi tmp, 0b00000001
        out LED_DDR, tmp
        out LED_PORT, tmp

        ldi tmp, 0b00000001
        out BUTT_PORT, tmp
        ldi tmp, 0x00
        out BUTT_DDR, tmp

        ldi tmp, 0b11111111
main:
        sbis pinb, 0
                rcall aktivnost
        rcall delay
        out LED_PORT, tmp

rjmp main

delay:
        clr r19
        clr r20
        ldi r21, cnt
        delay_loop:
                dec r19
                brne delay_loop
                dec r20
                brne delay_loop
                dec r21
                brne delay_loop
ret
aktivnost:
        com tmp
ret;
